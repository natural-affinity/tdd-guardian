def display_pattern_status(util, guards, errors)
	guards.each do | g |
		value = errors[g] == nil ? 'valid patterns' : nil
		util.display_status("guard-#{g}", value, "guard-#{g} #{errors[g]}", false)
	end unless guards.nil?
end
